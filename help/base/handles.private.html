# class User

需要事先接入MongoDB

import pymongo
client = pymongo.MongoClient('127.0.0.1', 27017)
db = client['qbot'] # 群数据库
userdb = client['user'] # 与网站对接

如需关闭网站对接 删除如下代码即可

userdb = client['user']
...
user = userdb.userdata.find_one({'qq': user_id})
self.user = None
if user is not None:
    self.user = user['user']

初始化需传入1个参数user_id

user = User(user_id)

完成后将会从数据库中获取信息好像__init__并没有使用async

作为一名QQ用户所拥有的数据并不会被获得，请在正常使用前调用传入数据

user.update_from_event(event)
